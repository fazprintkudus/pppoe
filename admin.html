Setup Google Sheets Integration

Ikuti langkah-langkah berikut untuk menghubungkan aplikasi dengan Google Spreadsheet

1

Buat Google Spreadsheet Baru

Buka Google Sheets dan buat spreadsheet baru untuk menyimpan data tagihan.

Buka Google Sheets
2

Buat Header Kolom di Baris Pertama

Copy header berikut dan paste di baris pertama spreadsheet:

Username | Name | PackageId | PackageName | Price | LastPay | Expired | Status | IP | Port

Pisahkan setiap kolom ke cell yang berbeda (A1, B1, C1, dst.)

3

Buka Apps Script Editor

Di spreadsheet, klik menu Extensions → Apps Script

Ini akan membuka editor kode Google Apps Script di tab baru
4

Paste Kode Script Berikut

Hapus semua kode yang ada, lalu paste kode berikut:

function doGet(e) {
  return handleRequest(e);
}

function doPost(e) {
  return handleRequest(e);
}

function handleRequest(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = sheet.getDataRange().getValues();
  var action = e.parameter.action;
  
  var output = ContentService.createTextOutput();
  output.setMimeType(ContentService.MimeType.JSON);
  
  // UPDATE ACTION - Update or create new row
  if (action === 'update') {
    var username = e.parameter.username;
    var name = e.parameter.name;
    var packageId = e.parameter.packageId;
    var packageName = e.parameter.packageName;
    var price = e.parameter.price;
    var lastPay = e.parameter.lastPay;
    var expired = e.parameter.expired;
    var status = e.parameter.status;
    var ip = e.parameter.ip || '';
    var port = e.parameter.port || '';
    
    // Find existing row by username
    var found = false;
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] === username) {
        // Update existing row
        sheet.getRange(i + 1, 2).setValue(name);
        sheet.getRange(i + 1, 3).setValue(packageId);
        sheet.getRange(i + 1, 4).setValue(packageName);
        sheet.getRange(i + 1, 5).setValue(price);
        sheet.getRange(i + 1, 6).setValue(lastPay);
        sheet.getRange(i + 1, 7).setValue(expired);
        sheet.getRange(i + 1, 8).setValue(status);
        sheet.getRange(i + 1, 9).setValue(ip);
        sheet.getRange(i + 1, 10).setValue(port);
        found = true;
        output.setContent(JSON.stringify({
          success: true,
          action: 'updated',
          username: username
        }));
        return output;
      }
    }
    
    // If not found, create new row
    if (!found) {
      sheet.appendRow([
        username, name, packageId, packageName,
        price, lastPay, expired, status, ip, port
      ]);
      output.setContent(JSON.stringify({
        success: true,
        action: 'created',
        username: username
      }));
    }
    return output;
  }
  
  // GET ACTION - Return all data as JSON
  if (action === 'get') {
    var result = {};
    for (var i = 1; i < data.length; i++) {
      if (data[i][0]) {
        result[data[i][0]] = {
          name: data[i][1],
          packageId: data[i][2],
          packageName: data[i][3],
          price: data[i][4],
          lastPay: data[i][5],
          expired: data[i][6],
          status: data[i][7],
          ip: data[i][8],
          port: data[i][9]
        };
      }
    }
    output.setContent(JSON.stringify(result));
    return output;
  }
  
  // DELETE ACTION - Delete a row by username
  if (action === 'delete') {
    var usernameToDelete = e.parameter.username;
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] === usernameToDelete) {
        sheet.deleteRow(i + 1);
        output.setContent(JSON.stringify({
          success: true,
          action: 'deleted',
          username: usernameToDelete
        }));
        return output;
      }
    }
    output.setContent(JSON.stringify({
      success: false,
      error: 'Username not found'
    }));
    return output;
  }
  
  // DEFAULT - Return all data
  var result = {};
  for (var i = 1; i < data.length; i++) {
    if (data[i][0]) {
      result[data[i][0]] = {
        name: data[i][1],
        packageId: data[i][2],
        packageName: data[i][3],
        price: data[i][4],
        lastPay: data[i][5],
        expired: data[i][6],
        status: data[i][7],
        ip: data[i][8],
        port: data[i][9]
      };
    }
  }
  output.setContent(JSON.stringify(result));
  return output;
}

Deploy sebagai Web App

Ikuti langkah berikut untuk deploy:

  1. a Klik tombol Deploy di pojok kanan atas
  2. b Pilih New deployment
  3. c Klik ikon gear ⚙️ → pilih Web app
  4. d Set "Execute as" = Me
  5. e Set "Who has access" = Anyone
  6. f Klik Deploy dan copy URL yang diberikan
